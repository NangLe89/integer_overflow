572      /**
573       * Refills the input buffer.
574       *
575       * @return      <code>false</code>, iff there was new input.
576       * 
577       * @exception   java.io.IOException  if any I/O-Error occurs
578       */
579      private boolean zzRefill() throws java.io.IOException {
580    
581        /* first: make room (if you can) */
582        if (zzStartRead > 0) {
583          zzEndRead += zzFinalHighSurrogate;
584          zzFinalHighSurrogate = 0;
585          System.arraycopy(zzBuffer, zzStartRead,
586                           zzBuffer, 0,
587                           zzEndRead-zzStartRead);
588    
589          /* translate stored positions */
590          zzEndRead-= zzStartRead;
591          zzCurrentPos-= zzStartRead;
592          zzMarkedPos-= zzStartRead;
593          zzStartRead = 0;
594        }
595    
596        /* is the buffer big enough? */
597        if (zzCurrentPos >= zzBuffer.length - zzFinalHighSurrogate) {
598          /* if not: blow it up */
599          char newBuffer[] = new char[zzBuffer.length*2];
600          System.arraycopy(zzBuffer, 0, newBuffer, 0, zzBuffer.length);
601          zzBuffer = newBuffer;
602          zzEndRead += zzFinalHighSurrogate;
603          zzFinalHighSurrogate = 0;
604        }
605    
606        /* fill the buffer with new input */
607        int requested = zzBuffer.length - zzEndRead;
608        int numRead = zzReader.read(zzBuffer, zzEndRead, requested);
609    
610        /* not supposed to occur according to specification of java.io.Reader */
611        if (numRead == 0) {
612          throw new java.io.IOException("Reader returned 0 characters. See JFlex examples for workaround.");
613        }
614        if (numRead > 0) {
615          zzEndRead += numRead;
616          /* If numRead == requested, we might have requested to few chars to
617             encode a full Unicode character. We assume that a Reader would
618             otherwise never return half characters. */
619          if (numRead == requested) {
620            if (Character.isHighSurrogate(zzBuffer[zzEndRead - 1])) {
621              --zzEndRead;
622              zzFinalHighSurrogate = 1;
623            }
624          }
625          /* potentially more input available */
626          return false;
627        }
628    
629        /* numRead < 0 ==> end of stream */
630        return true;
631      }
